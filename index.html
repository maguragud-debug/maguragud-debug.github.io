// ===== КОНТЕКСТНОЕ МЕНЮ ===== const phoneLink = document.getElementById('phoneLink'); const contextMenu = document.getElementById('contextMenu'); const menuCopyLink = document.getElementById('menuCopyLink'); const backHomeBtn = document.getElementById('backHomeBtn'); const submitBtn = document.getElementById('submitBtn'); const PHONE_NUMBER = '+380671234567'; const TEL_LINK = 'tel:' + PHONE_NUMBER; // Функция показа меню function showMenu(x, y, linkToCopy) { contextMenu.classList.add('active'); contextMenu.style.left = x + 'px'; contextMenu.style.top = y + 'px'; contextMenu.setAttribute('data-link', linkToCopy); } // Функция скрытия меню function hideMenu() { contextMenu.classList.remove('active'); } // ===== ПРАВЫЙ КЛИК на номере телефона ===== if (phoneLink) { phoneLink.addEventListener('contextmenu', function(e) { e.preventDefault(); e.stopPropagation(); showMenu(e.clientX, e.clientY, TEL_LINK); }); } // ===== ПРАВЫЙ КЛИК на кнопке "На головну" ===== if (backHomeBtn) { backHomeBtn.addEventListener('contextmenu', function(e) { e.preventDefault(); e.stopPropagation(); const currentUrl = window.location.href; showMenu(e.clientX, e.clientY, currentUrl); }); } // ===== ПРАВЫЙ КЛИК на кнопке "Відправити заявку" ===== if (submitBtn) { submitBtn.addEventListener('contextmenu', function(e) { e.preventDefault(); e.stopPropagation(); // Копируем ссылку на страницу благодарности (с ?thanks=1) const baseUrl = window.location.href.split('?')[0]; const thanksUrl = baseUrl + '?thanks=1'; showMenu(e.clientX, e.clientY, thanksUrl); }); } // ===== КЛИК на пункте меню ===== menuCopyLink.addEventListener('click', function(e) { e.preventDefault(); e.stopPropagation(); const linkToCopy = contextMenu.getAttribute('data-link') || window.location.href; navigator.clipboard.writeText(linkToCopy).then(() => { alert('✅ Скопійовано: ' + linkToCopy); hideMenu(); }).catch(() => { const textarea = document.createElement('textarea'); textarea.value = linkToCopy; document.body.appendChild(textarea); textarea.select(); document.execCommand('copy'); document.body.removeChild(textarea); alert('✅ Скопійовано: ' + linkToCopy); hideMenu(); }); }); // Закрыть меню при клике вне его document.addEventListener('click', function(e) { if (!contextMenu.contains(e.target)) { hideMenu(); } }); document.addEventListener('keydown', function(e) { if (e.key === 'Escape') hideMenu(); }); document.addEventListener('scroll', function() { hideMenu(); }); window.addEventListener('resize', function() { hideMenu(); });
